Embedded GarageBand 1.0
The project is aimed at developing an embedded Garage band that uses an 8051 microcontroller. The user can access the instruments via the touchscreen+graphics LCD or use the instruments that we have constructed. In this document we will cover the complete detail of the project including its construction, hardware design, software design and debugging.

00039 {
00040     register uint8 osc32k_bm = CLKCON & CLKCON_OSC32K_BM;
00041 
00042     // Source can have the following values:
00043     // CLOCK_SRC_XOSC   0x00  High speed Crystal Oscillator (XOSC)
00044     // CLOCK_SRC_HFRC   0x01  Low power RC Oscillator (HFRC)
00045 
00046     if (source == CLOCK_SRC_HFRC)
00047     {
00048         SLEEP &= ~SLEEP_OSC_PD_BM;       // power up both oscillators
00049         while (!CC2430_IS_HFRC_STABLE());// wait until the oscillator is stable
00050 
00051         asm("NOP");
00052         CLKCON = (osc32k_bm | CLKCON_OSC_BM | TICKSPD_DIV_1 | CLKCON_CLKSPD_BM);
00053         while (CLKCON != (osc32k_bm | CLKCON_OSC_BM | TICKSPD_DIV_1 | CLKCON_CLKSPD_BM));
00054 
00055         SLEEP |= SLEEP_OSC_PD_BM;        // power down the unused oscillator
00056     }
00057     else if (source == CLOCK_SRC_XOSC)
00058     {
00059         SLEEP &= ~SLEEP_OSC_PD_BM;       // power up both oscillators
00060         while (!CC2430_IS_XOSC_STABLE());// wait until the XOSC is stable
00061 
00062         asm("NOP");
00063         halMcuWaitUs(64);
00064         CLKCON = (osc32k_bm | TICKSPD_DIV_1);
