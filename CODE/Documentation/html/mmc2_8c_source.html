Embedded GarageBand 1.0
The project is aimed at developing an embedded Garage band that uses an 8051 microcontroller. The user can access the instruments via the touchscreen+graphics LCD or use the instruments that we have constructed. In this document we will cover the complete detail of the project including its construction, hardware design, software design and debugging.

00131 //Function      : to send a command to SD card
00132 //Arguments     : unsigned char (8-bit command value)
00133 //                        & unsigned long (32-bit command argument)
00134 //return        : unsigned char; response byte
00135 //******************************************************************
00136 unsigned char SD_sendCommand(unsigned char cmd, unsigned long arg)
00137 {
00138   unsigned char response, retry=0, status;
00139   
00140   //SD card accepts byte address while SDHC accepts block address in multiples of 512
00141   //so, if it's SD card we need to convert block address into corresponding byte address by 
00142   //multipying it with 512. which is equivalent to shifting it left 9 times
00143   //following 'if' loop does that
00144   
00145   if(SDHC_flag == 0)            
00146     if(cmd == READ_SINGLE_BLOCK     ||
00147        cmd == READ_MULTIPLE_BLOCKS  ||
00148          cmd == WRITE_SINGLE_BLOCK    ||
00149            cmd == WRITE_MULTIPLE_BLOCKS ||
00150              cmd == ERASE_BLOCK_START_ADDR|| 
00151                cmd == ERASE_BLOCK_END_ADDR ) 
00152     {
00153       arg = arg << 9;
00154     }      
00155   
00156   SD_CS_ASSERT;
00157   
00158   SPI_transmit(cmd | 0x40); //send command, first two bits always '01'
00159   SPI_transmit(arg>>24);
00160   SPI_transmit(arg>>16);
00161   SPI_transmit(arg>>8);
00162   SPI_transmit(arg);
00163   
00164   if(cmd == SEND_IF_COND)        //it is compulsory to send correct CRC for CMD8 (CRC=0x87) & CMD0 (CRC=0x95)
00165     SPI_transmit(0x87);    //for remaining commands, CRC is ignored in SPI mode
00166   else 
00167     SPI_transmit(0x95); 
00168   
00169   while((response = SPI_transmit(0xFF)) == 0xff) //wait response
00170     if(retry++ > 0xfe) break; //time out error
00171   
00172   if(response == 0x00 && cmd == 58)  //checking response of CMD58
00173   {
00174     status = SPI_transmit(0xFF) & 0x40;     //first byte of the OCR register (bit 31:24)
00175     if(status == 0x40) SDHC_flag = 1;  //we need it to verify SDHC card
00176     else SDHC_flag = 0;
00177     
00178     SPI_transmit(0xFF); //remaining 3 bytes of the OCR register are ignored here
00179     SPI_transmit(0xFF); //one can use these bytes to check power supply limits of SD
00180     SPI_transmit(0xFF); 
00181   }
00182   
00183   SPI_transmit(0xFF); //extra 8 CLK
00184   SD_CS_DEASSERT;
00185   
00186   return response; //return state
00187 }
